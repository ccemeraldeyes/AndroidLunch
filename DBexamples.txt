// examples of how to use android sqlite database with weShould



private void example(){
	WSdb db;
	Log.v("TestActivity.onCreate","Creating db in OnCreate");
	db=new WSdb(this);
	db.open();
	        
// rebuild test drops the tables, rebuilds them and fills with test data
// it is implemented in WSdb.java if you want to add or remove test data
	db.rebuildTest();
	        
// get all items in list
	Cursor c = db.getAllItems();
	    	
	// database must remain open while using cursor
		
	// how many items were returned?
	int x = c.getCount();
	    	
	// print message to logcat
	Log.v("cursor count","found " + x + " items");
			
	// iterate through list and display results using toast
	// move to top of list and ensure there is a result
	if (c.moveToFirst()){
		do {
	    			
	    // method to display Item using toast.  
	    // see below for example of how to access fields returned
	    DisplayItem(c);
	    			
	    } while (c.moveToNext());
	}
	    	
	// be sure to close the cursor
	c.close();
	    	
	    	
	// delete item by sending its id
	db.deleteItem(2);

	// get single item by id
	c = db.getItem(1);
	
	// get all categories	    	
	c = db.getAllCategories();
	   
	    
	// get all tags	    	
	c = db.getAllTags();
	    
	    	
	// get items with specific tag tagid=1	    	
	c=db.getItemsOfTag(1);
	   
	    	
	// ** close the database   	
	db.close();
}
	    	
	    	
// be sure to close the db as soon as you are done,  
//if new activity is started and it uses the db, it will crash
// in general, the steps are...
// open db
// do query
// use cursor
// close cursor
// close db -- db cannot be closed until cursor is done being used
	    	
	    	
// using toast to display the items
	

// when you get the cursor, the fields are in the order they are created in the table.
// for Item: id, name, category id, mappable, Data(json code);
	public void DisplayItem(Cursor c){
		Toast.makeText(this, "ID: " + c.getInt(0) + "\n" +
						 "Name: " + c.getString(1) + "\n" +
						 "CategoryID: " + c.getInt(2) + "\n" +
						 "Data: " + c.getString(3),
						 Toast.LENGTH_LONG).show();
	}
	
	public void DisplayCategory(Cursor c){
		Toast.makeText(this, "ID: " + c.getInt(0) + "\n" +
						 "Name: " + c.getString(1) + "\n" +
						 "Color: " + c.getInt(2) + "\n" +
						 "Schema: " + c.getString(3) + "\n",
						 Toast.LENGTH_LONG).show();
	}
	
	public void DisplayTag(Cursor c){
		Toast.makeText(this, "ID: " + c.getInt(0) + "\n" +
						 "Name: " + c.getString(1) + "\n",// +
						// "Color: " + c.getInt(2) + "\n",
						 Toast.LENGTH_LONG).show();
	}
